lead_times_query = f"""
WITH events_all AS (
  SELECT
    e.car_id, e.name, e.time,
    JSON_VALUE(e.snapshot, '$.oem') AS oem,
    JSON_VALUE(e.snapshot, '$.infleeting_compound_id') AS compound_id
  FROM `datawarehouse-304513.cars_ops_postgres_public.car_events` e
  WHERE e.name IN (
    'car_ordered','car_state_changed_produced','car_state_changed_arrived_from_supplier',
    'car_state_changed_tech_prep_done','car_state_changed_ready_to_deliver','car_state_changed_in_subscription'
  )
),
cars_active_2025 AS (
  SELECT DISTINCT car_id FROM events_all
  WHERE time >= '2025-01-01' AND time < '2026-01-01'
),
filtered AS (
  SELECT e.* FROM events_all e
  INNER JOIN cars_active_2025 c ON e.car_id = c.car_id
),
state_changes AS (
  SELECT car_id, name, time, oem, compound_id,
    LAG(time) OVER (PARTITION BY car_id ORDER BY time) AS prev_time
  FROM filtered
),
with_days AS (
  SELECT *, DATE_DIFF(DATE(time), DATE(prev_time), DAY) AS days_in_status
  FROM state_changes WHERE prev_time IS NOT NULL
),
compound_lookup AS (
  SELECT DISTINCT car_id, compound_id FROM filtered
  WHERE name = 'car_state_changed_arrived_from_supplier' AND compound_id IS NOT NULL
)
SELECT
  d.oem AS brand,
  COALESCE(cl.compound_id, d.compound_id) AS compound,
  COUNT(DISTINCT d.car_id) AS cars,

  ROUND(AVG(IF(d.name='car_state_changed_produced', d.days_in_status, NULL)),1) AS avg_ordered_to_produced,
  APPROX_QUANTILES(IF(d.name='car_state_changed_produced', d.days_in_status, NULL), 100 IGNORE NULLS)[OFFSET(50)] AS med_ordered_to_produced,

  ROUND(AVG(IF(d.name='car_state_changed_arrived_from_supplier', d.days_in_status, NULL)),1) AS avg_produced_to_arrived,
  APPROX_QUANTILES(IF(d.name='car_state_changed_arrived_from_supplier', d.days_in_status, NULL), 100 IGNORE NULLS)[OFFSET(50)] AS med_produced_to_arrived,

  ROUND(AVG(IF(d.name='car_state_changed_tech_prep_done', d.days_in_status, NULL)),1) AS avg_arrived_to_techprep,
  APPROX_QUANTILES(IF(d.name='car_state_changed_tech_prep_done', d.days_in_status, NULL), 100 IGNORE NULLS)[OFFSET(50)] AS med_arrived_to_techprep,

  ROUND(AVG(IF(d.name='car_state_changed_ready_to_deliver', d.days_in_status, NULL)),1) AS avg_techprep_to_ready,
  APPROX_QUANTILES(IF(d.name='car_state_changed_ready_to_deliver', d.days_in_status, NULL), 100 IGNORE NULLS)[OFFSET(50)] AS med_techprep_to_ready,

  ROUND(AVG(IF(d.name='car_state_changed_in_subscription', d.days_in_status, NULL)),1) AS avg_ready_to_sub,
  APPROX_QUANTILES(IF(d.name='car_state_changed_in_subscription', d.days_in_status, NULL), 100 IGNORE NULLS)[OFFSET(50)] AS med_ready_to_sub

FROM with_days d
LEFT JOIN compound_lookup cl ON d.car_id = cl.car_id
WHERE (d.oem, COALESCE(cl.compound_id, d.compound_id)) IN ({filter_tuples})
GROUP BY 1, 2
ORDER BY brand, compound
"""

def car_deep_dive(car_id: str):
    """Fetch and display the full infleeting timeline for a single car."""
    query = f"""
    WITH state_changes AS (
      SELECT
        car_id, name, time,
        JSON_VALUE(snapshot, '$.oem') AS oem,
        JSON_VALUE(snapshot, '$.model') AS model,
        JSON_VALUE(snapshot, '$.infleeting_compound_id') AS compound_id,
        JSON_VALUE(snapshot, '$.status') AS status,
        LAG(time) OVER (ORDER BY time) AS prev_time,
        LAG(name) OVER (ORDER BY time) AS prev_name
      FROM `datawarehouse-304513.cars_ops_postgres_public.car_events`
      WHERE car_id = '{car_id}'
        AND name IN (
          'car_ordered','car_state_changed_produced','car_state_changed_arrived_from_supplier',
          'car_state_changed_tech_prep_done','car_state_changed_ready_to_deliver','car_state_changed_in_subscription'
        )
    )
    SELECT
      name AS event,
      time,
      oem, model, compound_id, status,
      DATE_DIFF(DATE(time), DATE(prev_time), DAY) AS days_in_prev_status
    FROM state_changes
    ORDER BY time
    """
    result = run_query(query)
    if len(result) == 0:
        print(f'No state change events found for car {car_id}')
        return None

    oem = result.iloc[0]['oem']
    model = result.iloc[0]['model']
    compound = result.iloc[-1]['compound_id']
    print(f'Car {car_id}: {oem} {model} @ {compound}')
    print(f'Total: {(result["time"].max() - result["time"].min()).days}d from ordered to {result.iloc[-1]["status"]}')
    print()
    return result[['event', 'time', 'days_in_prev_status', 'status']]

9. Expected vs. Actual Lead Times — Backtesting Accuracy¶

How the API actually calculates Available From¶

The ops-cars API (infleeting-lead-times.ts) uses a two-layer system:

Available From = start_date + compound_lead_time_days
Car Status start_date compound_lead_time Owner
ordered / produced max(Fleet ETA, today) OrderedLeadTimes / ProducedLeadTimes Fleet (ETA) + Infleeting (compound)
arrived_from_supplier today (or anchor date) ArrivedFromSupplierLeadTimes Infleeting
tech_prep_done today (or anchor date) TechPreparationDoneLeadTimes Infleeting
ready_to_deliver today (or anchor date) ReadyToDeliverLeadTimes Infleeting

Two independent accuracy questions¶

  1. Fleet Layer (ETA accuracy): How long does ordered → arrived_from_supplier actually take? Fleet predicts this via OEM Delivery Plans. We don't have per-car ETA snapshots in BigQuery, but we CAN measure the actual duration distribution.

  2. Infleeting Layer (compound accuracy): How accurate is t_arrived (predicted days from compound arrival → subscription)? We CAN directly compare this against actual arrived → in_subscription medians.

Why ordered = produced for most brands¶

For nearly every brand, t_ordered = t_produced (identical compound lead times). This means the API assumes 0 days of compound lead time difference between ordered and produced — the real time between these states is entirely determined by Fleet ETA. Only Kia (56→42d) and Nissan (35→21d) have different values.

9.1 Fleet Layer — Ordered→Arrived Duration (Fleet-owned, no ETA data available)¶

The Fleet team owns the ETA prediction: how long from ordered (or produced) until the car physically arrives at the compound (arrived_from_supplier). We cannot directly measure ETA accuracy because per-car ETA snapshots aren't in car_events. But we CAN show:

  1. Actual median days for ordered→arrived per brand/compound (how long Fleet's phase actually takes)
  2. Variance (how predictable is the Fleet phase?)
  3. Which brands dominate total prediction error via the Fleet phase

9.4 Per-Phase Accuracy (derived phase durations from t-minus differences)¶

Ownership split:

  • Ordered→Produced and Produced→Arrived: Fleet-owned — derived expected = 0 for most brands because the API relies on Fleet ETA, not compound lead times

10. Lifecycle State Skip Analysis¶

Some brands routinely skip lifecycle states (e.g., tech_prep_done for MG, BYD, BMW/MINI). When a state is skipped, the hardcoded lead time for that phase is meaningless — cars jump directly to the next state, collapsing two phases into one.

This section quantifies skip rates per brand/compound to identify where predictions should be adjusted or states removed from the model.

11. Distribution Analysis — Percentile Windows per Phase¶

Median alone hides the spread. To build calibrated windows that hit 95-99% accuracy, we need the full distribution: P5, P25, P50, P75, P95, P99.

This reveals:

  • Tight distributions (P5 ≈ P95): predictable, safe to use median
  • Wide distributions (P95 >> P50): high variance, need wider windows or noise filtering
  • Heavy right tails (P99 >> P95): outliers from financing blockers, call-off delays, etc.

pctl_query = f"""
WITH events_all AS (
  SELECT
    e.car_id, e.name, e.time,
    JSON_VALUE(e.snapshot, '$.oem') AS oem,
    JSON_VALUE(e.snapshot, '$.infleeting_compound_id') AS compound_id
  FROM `datawarehouse-304513.cars_ops_postgres_public.car_events` e
  WHERE e.name IN (
    'car_ordered','car_state_changed_produced','car_state_changed_arrived_from_supplier',
    'car_state_changed_tech_prep_done','car_state_changed_ready_to_deliver','car_state_changed_in_subscription'
  )
),
cars_active_2025 AS (
  SELECT DISTINCT car_id FROM events_all
  WHERE time >= '2025-01-01' AND time < '2026-01-01'
),
filtered AS (
  SELECT e.* FROM events_all e
  INNER JOIN cars_active_2025 c ON e.car_id = c.car_id
),
state_changes AS (
  SELECT car_id, name, time, oem, compound_id,
    LAG(time) OVER (PARTITION BY car_id ORDER BY time) AS prev_time
  FROM filtered
),
with_days AS (
  SELECT *, DATE_DIFF(DATE(time), DATE(prev_time), DAY) AS days_in_status
  FROM state_changes WHERE prev_time IS NOT NULL
),
compound_lookup AS (
  SELECT DISTINCT car_id, compound_id FROM filtered
  WHERE name = 'car_state_changed_arrived_from_supplier' AND compound_id IS NOT NULL
),
enriched AS (
  SELECT
    d.car_id, d.name, d.days_in_status, d.oem AS brand,
    COALESCE(cl.compound_id, d.compound_id) AS compound
  FROM with_days d
  LEFT JOIN compound_lookup cl ON d.car_id = cl.car_id
  WHERE (d.oem, COALESCE(cl.compound_id, d.compound_id)) IN ({filter_tuples})
)
SELECT
  brand, compound,
  CASE name
    WHEN 'car_state_changed_produced' THEN 'Ordered→Produced'
    WHEN 'car_state_changed_arrived_from_supplier' THEN 'Produced→Arrived'
    WHEN 'car_state_changed_tech_prep_done' THEN 'Arrived→Tech Prep'
    WHEN 'car_state_changed_ready_to_deliver' THEN 'Tech Prep→Ready'
    WHEN 'car_state_changed_in_subscription' THEN 'Ready→Subscription'
  END AS phase,
  COUNT(*) AS n,
  APPROX_QUANTILES(days_in_status, 100 IGNORE NULLS)[OFFSET(5)] AS p5,
  APPROX_QUANTILES(days_in_status, 100 IGNORE NULLS)[OFFSET(25)] AS p25,
  APPROX_QUANTILES(days_in_status, 100 IGNORE NULLS)[OFFSET(50)] AS p50,
  APPROX_QUANTILES(days_in_status, 100 IGNORE NULLS)[OFFSET(75)] AS p75,

12. Granularity Comparison — Brand-Level vs. Brand+Compound-Level¶

13. Data-Driven Lead Times & Calibrated Windows¶

Replace static hardcoded values with data-driven lead times and calibrated deviation windows that target 95% accuracy.

For each brand/compound/phase:

  • Point estimate: P50 (median) — best single guess
  • Optimistic bound: P25 — "if things go well"
  • Conservative bound: P75 — "realistic buffer"
  • 95% window: [P5, P95] — should capture 90% of cars
  • 99% window: [P1, P99] — should capture 98% of cars

This produces a ready-to-use replacement table for the current static lead times.

14. Blocker Performance vs Configured Shift Times¶

Goal: Compare actual blocker resolution times against configured criticality thresholds (from isCritical in check-infleeting-blockers.ts).

  • Red = too slow: Blockers take longer to resolve than the configured threshold → late delivery risk

Configured Criticality Thresholds¶

These values come from check-infleeting-blockers.ts — the isCritical(blockedDays, handoverDate, N) calls. Each blocker becomes critical when there are ≤ N working days until the handover date.

We use these as the benchmark: if a blocker typically takes longer than N working days to resolve, the threshold is too short and cars will be delayed.

Core Query — Blocker Episode Duration¶

The blockers_events table logs individual events: blocker_created, blocker_resolved, blocker_deleted. A single car can have multiple episodes of the same blocker (e.g., PDI not done → resolved → created again).

Pairing logic: For each (car_id, blocking_reason), we use ROW_NUMBER() to match the 1st blocker_created with the 1st blocker_resolved/blocker_deleted, the 2nd with the 2nd, etc. This gives us the duration of each individual blocker episode.

Working days approximation: calendar_days − 2 × floor(calendar_days / 7) — subtracts weekends but not holidays. Sufficient for comparative analysis.

Filters: 2025 data only, durations 0–365 days (excludes negative/stale), restricted to the 40 brand/compound combos defined above.

In [ ]:
blocker_query = f"""
WITH blocker_pairs AS (
  -- Pair Nth blocker_created with Nth blocker_resolved/blocker_deleted per (car_id, blocking_reason)
  SELECT
    c.car_id,
    c.blocking_reason,
    c.time AS created_time,
    r.time AS resolved_time,
    r.name AS resolution_type,
    DATE_DIFF(DATE(r.time), DATE(c.time), DAY) AS duration_calendar_days
  FROM (
    SELECT car_id, blocking_reason, time,
      ROW_NUMBER() OVER (PARTITION BY car_id, blocking_reason ORDER BY time) AS rn
    FROM `datawarehouse-304513.cars_ops_postgres_public.blockers_events`
    WHERE name = 'blocker_created'
      AND time >= '2025-01-01' AND time < '2026-01-01'
  ) c
  INNER JOIN (
    SELECT car_id, blocking_reason, name, time,
      ROW_NUMBER() OVER (PARTITION BY car_id, blocking_reason ORDER BY time) AS rn
    FROM `datawarehouse-304513.cars_ops_postgres_public.blockers_events`
    WHERE name IN ('blocker_resolved', 'blocker_deleted')
      AND time >= '2025-01-01' AND time < '2026-01-01'
  ) r ON c.car_id = r.car_id
    AND c.blocking_reason = r.blocking_reason
    AND c.rn = r.rn
),
-- Get brand/compound from car_events snapshot (same approach as existing notebook)
car_brand_compound AS (
  SELECT DISTINCT
    car_id,
    JSON_VALUE(snapshot, '$.oem') AS brand,
    JSON_VALUE(snapshot, '$.infleeting_compound_id') AS compound
  FROM `datawarehouse-304513.cars_ops_postgres_public.car_events`
  WHERE name = 'car_state_changed_arrived_from_supplier'
    AND JSON_VALUE(snapshot, '$.oem') IS NOT NULL
    AND JSON_VALUE(snapshot, '$.infleeting_compound_id') IS NOT NULL
)
SELECT
  bp.car_id,
  bp.blocking_reason,
  bp.created_time,
  bp.resolved_time,
  bp.resolution_type,
  bp.duration_calendar_days,
  -- Approximate working days: calendar days minus weekends (2 per full week)
  bp.duration_calendar_days - 2 * DIV(bp.duration_calendar_days, 7) AS duration_working_days_approx,
  cbc.brand,
  cbc.compound
FROM blocker_pairs bp
INNER JOIN car_brand_compound cbc ON bp.car_id = cbc.car_id
WHERE bp.duration_calendar_days >= 0
  AND bp.duration_calendar_days <= 365
  AND (cbc.brand, cbc.compound) IN ({filter_tuples})
ORDER BY bp.car_id, bp.blocking_reason, bp.created_time
"""

Aggregation¶

Group blocker episodes by brand/compound/blocker type and compute:

  • Median & Mean working-day duration — central tendency
  • P75 — the "1 in 4 cars takes longer than this" threshold, used for recommendations
  • P95 — extreme cases
  • Delta columns: actual − configured — positive = blocker takes longer than threshold (late risk)

Also computes a fleet-wide aggregation (all brand/compounds combined) for the summary view.

# Bucket days_before_pickup
def bucket_days(d):
    if pd.isna(d):
        return None
    d = int(d)
    if d <= 1: return '0-1d'
    if d <= 3: return '2-3d'
    if d <= 7: return '4-7d'
    if d <= 14: return '8-14d'
    return '15+d'

pred_df['days_before_bucket'] = pred_df['days_before_pickup'].apply(bucket_days)
pred_valid = pred_df.dropna(subset=['days_before_bucket'])

    median_delay_cd=('duration_calendar_days', 'median'),
    p75_delay_cd=('duration_calendar_days', lambda x: np.percentile(x, 75)),
    count=('car_id', 'count')
).reset_index()

# Interactive lookup function
def predict_delay(compound_label, blocker, days_before):
    bucket = bucket_days(days_before)
    match = pred_agg[
        (pred_agg['label'] == compound_label) &
        (pred_agg['blocking_reason'] == blocker) &
        (pred_agg['days_before_bucket'] == bucket)
    ]
    if len(match) == 0:
        # Fallback: all compounds for this blocker/bucket
        fallback = pred_agg[
            (pred_agg['blocking_reason'] == blocker) &
            (pred_agg['days_before_bucket'] == bucket)
        ]
        if len(fallback) == 0:
            return "No data available for this combination"
        med = fallback['median_delay_cd'].median()
        p75 = fallback['p75_delay_cd'].median()
        return f"Fleet-wide estimate (no compound-specific data): median {med:.0f}d, P75 {p75:.0f}d (from {fallback['count'].sum()} episodes)"
    row = match.iloc[0]
    return f"Expected delay: median {row['median_delay_cd']:.0f}d, P75 {row['p75_delay_cd']:.0f}d (from {row['count']} episodes)"

15. Pickup Too Soon — Root Cause & Early Warning Analysis¶

pickup_too_soon fires when a car's planned pickup date is imminent but the delivery is still in DRAFT status — meaning the car isn't ready. This section investigates:

  1. Which other blockers were still active when pickup_too_soon fired — these are the root causes preventing readiness
  2. Which blockers appeared days/weeks before pickup_too_soon — these are potential early warning signals that could trigger proactive intervention
  3. Which brand/compound combinations are most affected — where to focus operational improvements

Key question: Do we already have the data signals to predict pickup_too_soon earlier, or do we need new ones?

15.1 Root Causes — Blockers Active When Pickup Too Soon Fired¶

For each pickup_too_soon episode, we check which other blockers were still unresolved at that moment. These are the actual reasons the car wasn't ready.

We also look at precursor blockers — any blocker created in the 30 days before pickup_too_soon, even if already resolved. These show the operational "journey" that led to the delay.

15.3 Early Warning Signals — How Far Ahead Could We Have Known?¶

For cars that eventually got pickup_too_soon, we look at all blockers created at any time before the first PTS event. The earlier a blocker appears, the more lead time it gives for proactive intervention.

Interpretation: If pdi_not_done appears median 110 days before pickup_too_soon on 1,388 cars — we had 110 days of warning that these cars might not be ready in time.

In [ ]:
# Early warning: ALL blockers that appeared before first pickup_too_soon per car
early_warning_query = f"""
WITH
pts_cars AS (
  SELECT car_id, MIN(time) AS first_pts_time
  FROM `datawarehouse-304513.cars_ops_postgres_public.blockers_events`
  WHERE name = 'blocker_created'
    AND blocking_reason = 'pickup_too_soon'
    AND time >= '2025-01-01' AND time < '2026-01-01'
  GROUP BY car_id
),
car_brand_compound AS (
  SELECT DISTINCT car_id,
    JSON_VALUE(snapshot, '$.oem') AS brand,
    JSON_VALUE(snapshot, '$.infleeting_compound_id') AS compound
  FROM `datawarehouse-304513.cars_ops_postgres_public.car_events`
  WHERE name = 'car_state_changed_arrived_from_supplier'
    AND JSON_VALUE(snapshot, '$.oem') IS NOT NULL
    AND JSON_VALUE(snapshot, '$.infleeting_compound_id') IS NOT NULL
),
early_blockers AS (
  SELECT be.car_id, be.blocking_reason, be.time AS blocker_time,
    pts.first_pts_time,
    DATE_DIFF(DATE(pts.first_pts_time), DATE(be.time), DAY) AS days_before_pts,
    cbc.brand, cbc.compound
  FROM `datawarehouse-304513.cars_ops_postgres_public.blockers_events` be
  INNER JOIN pts_cars pts ON be.car_id = pts.car_id
  INNER JOIN car_brand_compound cbc ON be.car_id = cbc.car_id
  WHERE be.name = 'blocker_created'
    AND be.blocking_reason NOT IN ('return_not_scheduled', 'pickup_too_soon', 'pickup_date_in_the_past',
      'embezzlement', 'ongoing_sales_negotiations', 'temporary_return_block', 'potential_sale_to_customer',
      'registered', 'winter_tires', 'appraisal_missing', 'zb2_location', 'service_missing', 'damages_not_processed', 'warranty_reactivation')
    AND be.time < pts.first_pts_time
    AND be.time >= '2025-01-01'
    AND (cbc.brand, cbc.compound) IN ({filter_tuples})
)
SELECT blocking_reason, brand, compound,
  COUNT(*) AS episodes,
  COUNT(DISTINCT car_id) AS unique_cars,
  APPROX_QUANTILES(days_before_pts, 100)[OFFSET(25)] AS p25_days_early,
  APPROX_QUANTILES(days_before_pts, 100)[OFFSET(50)] AS median_days_early,
  APPROX_QUANTILES(days_before_pts, 100)[OFFSET(75)] AS p75_days_early
FROM early_blockers
GROUP BY blocking_reason, brand, compound
"""
